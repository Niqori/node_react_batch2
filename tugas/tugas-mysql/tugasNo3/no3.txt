SELECT
    a.id,
    a.title,
    a.body,
    u.name AS author_name
FROM
    blog_posts a
JOIN
    users u ON a.author_id = u.id
WHERE
    u.name = 'John Doe';

NO 3 B.


SELECT
    c.name AS category_name,
    bp.title AS post_title,
    u.name AS author_name
FROM
    blog_posts bp
JOIN
    users u ON bp.author_id = u.id
JOIN
    post_categories pc ON bp.id = pc.post_id
JOIN
    categories c ON pc.category_id = c.id
WHERE
    c.name = 'Basic'
    AND bp.id IN (
        SELECT post_id
        FROM post_categories
        GROUP BY post_id
        HAVING COUNT(category_id) = 1
    );

NO 3 C.

SELECT
    c.name AS category_name,
    bp.title AS post_title,
    bp.body AS post_body,
    u.name AS author_name
FROM
    blog_posts bp
JOIN
    users u ON bp.author_id = u.id
JOIN
    post_categories pc ON bp.id = pc.post_id
JOIN
    categories c ON pc.category_id = c.id
WHERE
    u.name IN ('Robert', 'Frank');

NO 3 D.

SELECT DISTINCT
    u.name AS author_name,
    c.name AS category_name
FROM
    blog_posts bp
JOIN
    users u ON bp.author_id = u.id
JOIN
    post_categories pc ON bp.id = pc.post_id
JOIN
    categories c ON pc.category_id = c.id
WHERE
    bp.id IN (
        SELECT post_id
        FROM post_categories
        GROUP BY post_id
        HAVING COUNT(category_id) >= 2
    );
